# 界面展示的新思路

  • createApp 创建组件实例和mount方法挂载展示的新途径
import {createApp} from 'vue'
import alert from './alert.vue'
type alertProps = 'error' | 'success' | 'primary' | 'warning'
const createAlert = (type: alertProps, message: string, timeout = 2000) => {
    const app = createApp(alert, {
        message,
        type
    })
    const alertEl = document.createElement('div')
    document.body.appendChild(alertEl)
    app.mount(alertEl)
    setTimeout(() => {
        app.unmount(alertEl)
        document.body.removeChild(alertEl)
    }, timeout)
}
export default createAlert

# vue3 实现自定义组件的v-model

  • 自定义组件中定义一个modelValue 属性
  • 监听@input事件 context.emit('update:modelValue', e.target.value)
  • 完成前两步后组件使用v-model即可
export default defineComponent({
    props: {
        modelValue: {
            type: String
        }
    }
     setup(props, context){
         const formElement = reactive({
            val: props.modelValue ||  '',
            inputChange: (e: KeyboardEvent) =>{
                const targetValue = (e.target as HTMLInputElement).value
                formElement.val = targetValue
                context.emit('update:modelValue', targetValue)
            }
        })
        return {
            formElement
        }
     }
})
